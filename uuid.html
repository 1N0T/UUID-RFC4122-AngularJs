var fecha = new Date().getTime();
var uuid  = 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g,
            function(c) {
               var r = (fecha + Math.random()*16)%16 | 0;
               return (c=='x' ? r : (r&0x7|0x8)).toString(16);
            });
return uuid;
